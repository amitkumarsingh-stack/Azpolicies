# Patch all mutating webhooks directly without counting
for webhook in $(kubectl get mutatingwebhookconfigurations | grep kyverno | awk '{print $1}'); do
  echo "Patching $webhook"
  kubectl get mutatingwebhookconfiguration $webhook -o json | \
    sed 's/"failurePolicy":"Fail"/"failurePolicy":"Ignore"/g' | \
    kubectl replace -f -
done

# Patch all validating webhooks
for webhook in $(kubectl get validatingwebhookconfigurations | grep kyverno | awk '{print $1}'); do
  echo "Patching $webhook"
  kubectl get validatingwebhookconfiguration $webhook -o json | \
    sed 's/"failurePolicy":"Fail"/"failurePolicy":"Ignore"/g' | \
    kubectl replace -f -
done
