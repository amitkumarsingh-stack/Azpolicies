# Patch all mutating webhooks - loop through all webhooks in each config
for webhook in $(kubectl get mutatingwebhookconfigurations | grep kyverno | awk '{print $1}'); do
  echo "Patching $webhook"
  count=$(kubectl get mutatingwebhookconfiguration $webhook \
    -o jsonpath='{.webhooks}' | python3 -c "import sys,json; print(len(json.load(sys.stdin)))")
  for i in $(seq 0 $((count-1))); do
    kubectl patch mutatingwebhookconfiguration $webhook \
      --type=json \
      -p="[{\"op\":\"replace\",\"path\":\"/webhooks/$i/failurePolicy\",\"value\":\"Ignore\"}]"
  done
done
