# Template the crds subchart directly
helm template kyverno-crds charts/crds \
  --set groups.wgpolicyk8s.policyreports=true \
  --set groups.wgpolicyk8s.clusterpolicyreports=true \
  --set groups.reports.ephemeralreports=true \
  --set groups.reports.clusterephemeralreports=true \
  --set reportsServer.enabled=false \
  | kubectl apply -f -
